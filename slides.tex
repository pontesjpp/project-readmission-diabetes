\section{Business Understanding}

\begin{frame}{Briefing do cliente}
\textbf{Cliente fictício:} SI5 Health, rede hospitalar com unidade de cuidado pós-alta.

\vspace{0.6em}
\textbf{Problema.} Identificar, na alta do paciente diabético, a probabilidade de \alert{readmissão precoce} (<30 dias) para acionar acompanhamento ativo.

\vspace{0.6em}
\textbf{Saída.} Classe $\in \{\texttt{<30}, \texttt{>30}, \texttt{NO}\}$.

\vspace{0.6em}
\textbf{Métrica de sucesso.} F1-macro no conjunto de teste (dataset desbalanceado; \texttt{<30} representa 11\%).

\vspace{0.6em}
\textbf{Restrições.} Falso negativo da classe \texttt{<30} é o erro mais caro (paciente readmite sem suporte). Limiar de operação calibrado pela capacidade do time de pós-alta.

\vspace{0.6em}
\textbf{Implicações éticas.} Viés demográfico (race, gender, age), privacidade (dados pseudonimizados), automação de decisão clínica como apoio, não substituição.
\end{frame}

\section{Data Understanding \& EDA}

\begin{frame}{Dataset}
Diabetes 130-US Hospitals (UCI 296, Strack et al., 2014).
\begin{itemize}
  \item 101.766 internações $\rightarrow$ 81.412 treino / 20.354 teste (split estratificado, \texttt{random\_state=42}).
  \item 47 features após remover IDs, \texttt{weight} (97\% missing), \texttt{examide}, \texttt{citoglipton}.
  \item Categóricas dominam: 24 medicações + raça, gênero, idade, diag\_1/2/3 (ICD-9 agrupado), \texttt{payer\_code}, \texttt{medical\_specialty} (alta cardinalidade).
  \item Faltantes codificados como \texttt{?} no original.
\end{itemize}
\end{frame}

  \item \texttt{number\_inpatient}, \texttt{number\_emergency} muito assimétricas $\rightarrow$ motivam variante \textbf{RobustScaler}.
  \item Missingness pesado em \texttt{race}, \texttt{payer\_code}, \texttt{medical\_specialty}, \texttt{diag\_*}.

\section{Data Preparation}

\begin{frame}{Pipeline baseline}
\textbf{Limpeza estrutural} (\texttt{RawCleaner}): \texttt{?} $\rightarrow$ NaN, drop de colunas inúteis, agrupamento ICD-9, IDs como string.

\vspace{0.5em}
\textbf{ColumnTransformer}:
\begin{itemize}
  \item Numéricas: \texttt{SimpleImputer(median)} $\rightarrow$ \texttt{StandardScaler}.
  \item Categóricas: \texttt{SimpleImputer(most\_frequent)} $\rightarrow$ \texttt{OneHotEncoder(handle\_unknown="ignore")}.
\end{itemize}

\vspace{0.5em}
\textbf{Crucial:} todo \texttt{fit\_transform} acontece \alert{dentro de cada fold} da CV. Nada toca o conjunto de teste antes da Seção 9.

\vspace{0.5em}
\textbf{Variantes:} \textbf{SMOTE} (\texttt{imblearn} pipeline), \textbf{RobustScaler}, \textbf{TargetEncoder} em colunas de alta cardinalidade.
\end{frame}

\begin{frame}{Contexto: F1 nesse dataset}
\textbf{Por que F1 $\sim$0.45 é competitivo?}

\vspace{0.5em}
\begin{itemize}
  \item Strack et al.\ (2014, paper original do dataset): \textbf{F1-macro $\sim$0.43}.
  \item Trabalhos posteriores: 0.42--0.48 com feature engineering pesada.
  \item Acima de 0.55: \alert{suspeito de leakage} (rótulo informa as features de algum jeito).
\end{itemize}

\vspace{0.6em}
\textbf{Por que é difícil?}
\begin{itemize}
  \item \texttt{<30} vs \texttt{>30} é uma fronteira \emph{clínica arbitrária}. Paciente que readmite em 28 dias é praticamente indistinguível de um que readmite em 32 dias.
  \item Forte desbalanceamento: \texttt{<30} = 11\%, \texttt{>30} = 35\%, \texttt{NO} = 54\%.
  \item Sinais clínicos compartilhados: idade, número de visitas anteriores, diagnósticos crônicos elevam risco \emph{nas duas} classes positivas.
\end{itemize}

\vspace{0.5em}
\textbf{Nossa contribuição não é F1 alto:} é uma comparação \emph{metodologicamente rigorosa} entre 10 algoritmos $\times$ 4 pré-processamentos, com Wilcoxon pareado para todas as decisões.
\end{frame}

\begin{frame}[shrink=10]{Achado metodológico: SMOTE não é gratuito}
\textbf{Padrão claro e simétrico entre SMOTE e Undersampling:}

\vspace{0.6em}
\begin{itemize}
  \item \textbf{Reamostragem ajuda} modelos \emph{sem balanceamento embutido}:
  \begin{itemize}
    \item LVQ: \textbf{+0.152} com undersample, +0.127 com SMOTE — \emph{maiores ganhos absolutos do estudo}
    \item Decision Tree: +0.018 com SMOTE, +0.010 com undersample
    \item LightGBM: +0.005 com SMOTE
  \end{itemize}
  \item \textbf{Reamostragem atrapalha} modelos com \texttt{class\_weight="balanced"} (selecionado pela busca HP): Random Forest perde 0.021 (SMOTE) / 0.018 (under); SVM perde 0.020 / 0.013. Ambas reamostragens \emph{deterioram} o classificador já calibrado.
\end{itemize}

\vspace{0.6em}
\textbf{Implicações.}
\begin{itemize}
  \item Em produção, escolher \emph{uma} estratégia de desbalanceamento e medir, não combinar.
  \item Custo computacional: SMOTE adiciona $\sim$30\% de linhas sintéticas em cada fold. Para LVQ valeu (ganho 50\%); para RF foi prejuízo de tempo \emph{e} F1.
\end{itemize}

\vspace{0.6em}
\textbf{RobustScaler:} efeito nulo em todos os modelos testados (p$\geq$0.05). Esperado: árvores e modelos baseados em ranking são invariantes a escala; SVM e MLP já normalizados pelo baseline (StandardScaler).

\vspace{0.4em}
\textbf{TargetEncoder é heterogêneo:} melhora DT e LVQ marginalmente (modelos sem representação sensível a valores), mas \alert{degrada K-NN (--0.010, p=0.002) e MLP (--0.014, p=0.032)} — modelos sensíveis à escala e representação das features. TargetEncoding em colunas de alta cardinalidade (\texttt{diag\_*}, \texttt{payer\_code}) introduz leakage controlado dentro do fold mas reduz granularidade vs OHE para modelos de distância/ativação.
\end{frame}

\begin{frame}[shrink=8]{Modelo vencedor}
\textbf{Vencedor:} \alert{Random Forest} (com pré-processamento baseline ou RobustScaler — \emph{empate técnico}).

\vspace{0.3em}
\textbf{F1-macro CV:} \textbf{0.4513 $\pm$ 0.0028} (5 folds estratificados).

\vspace{0.3em}
\textbf{Hiperparâmetros (encontrados na busca):}
\begin{itemize}\setlength{\itemsep}{0pt}
  \item \texttt{n\_estimators=200}, \texttt{max\_depth=20}, \texttt{max\_features="log2"}
  \item \texttt{min\_samples\_leaf=1}, \alert{\texttt{class\_weight="balanced"}}
\end{itemize}

\vspace{0.3em}
\textbf{Por quê esse vence:}
\begin{itemize}\setlength{\itemsep}{1pt}
  \item \textbf{Top do ranking} dentre as 29+ combinações avaliadas. SVM 2º (0.4471) — empate técnico ($\Delta$=0.004, Wilcoxon p=0.31).
  \item \texttt{class\_weight="balanced"} ataca o desbalanceamento \emph{na árvore}, sem custo de SMOTE.
  \item Curva treino vs val: overfit moderado ($F1_{\text{tr}}$=0.84, $F1_{\text{val}}$=0.45) mas \emph{estável} entre folds ($\sigma$=0.003).
  \item Inferência rápida ($\sim$ms/amostra), interpretabilidade via importância de features, pronto pra deployment.
  \item Variantes em RF: \textbf{SMOTE atrapalha} (--0.021, p$<$0.001), \emph{Robust empata}, TargetEnc atrapalha (--0.007).
\end{itemize}
\end{frame}

\section{Deployment}

\begin{frame}{Deployment e riscos}
\textbf{Uso.} Score acionado na alta; pacientes \texttt{<30} entram em fila de acompanhamento (ligação 48h, retorno 7d, revisão farmacoterapia).

\vspace{0.5em}
\textbf{Calibração.} Threshold tuning sobre \texttt{predict\_proba(<30)} para maximizar recall sujeito ao teto de fila operacional.

\vspace{0.5em}
\textbf{Riscos:}
\begin{itemize}
  \item Viés demográfico (race/age/gender). Mitigação: auditoria DIR por subgrupo, F1-macro estratificado mensal.
  \item Drift de população (nova unidade, mudança de mix). Mitigação: PSI semanal nas top-15 features.
  \item Privacidade: pseudonimização, sem retenção de raw fora de \texttt{data/raw/}.
\end{itemize}

\vspace{0.5em}
\textbf{Retreinamento.} Gatilhos: PSI>0.25 em $\geq$3 features, F1 mensal cai >0.05 vs teste, ou trimestralmente.
\end{frame}
